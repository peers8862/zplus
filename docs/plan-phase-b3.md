# Phase B3 — Derived Rendering (dashboards, Action Center, corpus.json) — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A `zplus gen-derived` step that turns the B2 corpus graph into viewable output — a dashboard table in each collection landing, a generated **Action Center** subpage under Mission Control, and an unpublished root `corpus.json` — all before `zensical build`, touching only tool-owned files.

**Architecture:** Pure builders in a new `render.py` (corpus → markdown/dict); an orchestrator `commands/derived.py` that writes dashboards into landing `index.md` files (marker-guarded region, mirroring `patch/toml_nav.py`), writes `docs/mission-control/action-center.md`, and writes `corpus.json` at project root. Wired into `serve`/`build` before `nav.regenerate` (so `gen-nav` navs the new subpage). Entry source files are never modified.

**Tech Stack:** Python ≥3.11, PyYAML, stdlib `json`, `unittest`.

## Global Constraints

- **Never modify hand-authored entry files.** `gen-derived` writes only: landing `index.md` files (a marker-guarded region — hand prose above it is preserved), the generated `mission-control/action-center.md`, and root `corpus.json`. Nothing else.
- **`corpus.json` is unpublished.** It is written to **project root** (outside `docs/`), so `zensical build` and staticrypt never ship it. It stays in the working tree (git-tracked) for local AI. Do **not** write it under `docs/`.
- **Action Center is a Mission Control subpage:** `docs/mission-control/action-center.md`. It is fully generated (overwritten each run) — no marker region, but a "generated by zplus" header comment. `gen-nav` lists it automatically (section pages, A→Z).
- **Idempotent.** Running `gen-derived` twice yields identical files. The dashboard region uses markers so re-runs replace, not append.
- **Ordering in the pipeline:** `gen_derived` runs **before** `nav.regenerate` in `serve` and `build` (the Action Center page must exist before `gen-nav` includes it).
- **Shapes/status membership only** (B2 rules). No status *workflow* logic, no boards, no mermaid graph — those are B4.
- Test framework: `unittest`; run `TMPDIR=/home/morgen/making/.zplus-scratch ./.venv/bin/python -m unittest discover -s tests`.

```python
"""Pure builders for derived content — corpus/manifest in, markdown/dict out.

No I/O. commands/derived.py does the file writing. Mirrors the marker-region
idea from patch/toml_nav.py, but with HTML comments for markdown.
"""
import re

DASH_BEGIN = "<!-- >>> ZPLUS_DASHBOARD — generated by zplus; edits below are overwritten >>> -->"
DASH_END = "<!-- <<< ZPLUS_DASHBOARD <<< -->"


def splice_md_region(text, begin, end, body):
    """Insert or replace a marker-delimited region. Idempotent."""
    region = begin + "\n\n" + body + "\n\n" + end
    if begin in text and end in text:
        pat = re.compile(re.escape(begin) + r".*?" + re.escape(end), re.S)
        return pat.sub(lambda _: region, text)
    sep = "" if text.endswith("\n") else "\n"
    return text + sep + "\n" + region + "\n"


def _as_list(value):
    if value is None or value == []:
        return []
    return value if isinstance(value, list) else [value]


def _fmt(value):
    return ", ".join(str(v) for v in _as_list(value))


def dashboard_table(doc_type, entries, corpus):
    """Markdown table: Entry link + one column per declared field + inbound count."""
    names = [f.name for f in doc_type.fields]
    header = ["Entry"] + names + ["Referenced by"]
    rows = ["| " + " | ".join(header) + " |",
            "| " + " | ".join(["---"] * len(header)) + " |"]
    for e in sorted(entries, key=lambda e: e.slug):
        inbound = sum(len(v) for v in e.backlinks.values())
        cells = [f"[{e.title}]({e.slug}.md)"]
        cells += [_fmt(e.fields.get(n)) for n in names]
        cells += [str(inbound) if inbound else ""]
        rows.append("| " + " | ".join(cells) + " |")
    return "\n".join(rows)
```

```python
OPEN_STATUS = {"proposed", "open", "building", "draft"}


def _bullet_lines(items):
    return "\n".join(items) if items else "_None._"


def action_center_markdown(corpus, m):
    """A generated worklist: unowned, open-status, and dangling-ref entries."""
    type_by_name = {t.name: t for t in m.types}
    unowned, open_items, dangling = [], [], []
    for e in corpus.entries:
        t = type_by_name.get(e.type_name)
        if not t:
            continue
        link = f"../{t.folder}/{e.slug}.md"
        ref = f"[{e.title}]({link})"
        if any(f.name == "owner" and f.required for f in t.fields) \
                and not e.fields.get("owner"):
            unowned.append(f"- {ref} — no owner")
        status = e.fields.get("status")
        if status in OPEN_STATUS:
            open_items.append(f"- {ref} — status `{status}`")
        for f in t.fields:
            if f.type != "ref":
                continue
            for slug in _as_list(e.fields.get(f.name)):
                if slug and slug not in corpus.by_slug:
                    dangling.append(f"- {ref} — `{f.name}` → missing `{slug}`")
    return (
        "<!-- generated by zplus gen-derived; edits are overwritten -->\n"
        "# Action Center\n\n"
        "What needs attention across the company, derived from every entry.\n\n"
        "## Needs an owner\n\n" + _bullet_lines(unowned) + "\n\n"
        "## Open / in progress\n\n" + _bullet_lines(open_items) + "\n\n"
        "## Broken links\n\n" + _bullet_lines(dangling) + "\n"
    )


def corpus_to_dict(corpus):
    return {"entries": [
        {"type": e.type_name, "slug": e.slug, "title": e.title,
         "fields": e.fields,
         "backlinks": {f"{k[0]}.{k[1]}": v for k, v in e.backlinks.items()}}
        for e in corpus.entries
    ]}
```

```python
"""gen-derived: write dashboards, the Action Center, and corpus.json from the corpus.

Runs before `zensical build`. Writes only tool-owned files — landing dashboards
(marker-guarded), docs/mission-control/action-center.md, and root corpus.json.
Never touches hand-authored entry files.
"""
import json
import os

from .. import manifest as manifest_mod, corpus as corpus_mod, render


def gen_derived(project_dir):
    m = manifest_mod.load(os.path.join(project_dir, "zplus.toml"))
    c = corpus_mod.resolve(corpus_mod.read_corpus(project_dir, m), m)
    docs = os.path.join(project_dir, "docs")

    by_type = {}
    for e in c.entries:
        by_type.setdefault(e.type_name, []).append(e)

    n_dash = 0
    for t in m.types:
        ents = by_type.get(t.name)
        landing = os.path.join(docs, t.folder, "index.md")
        if not ents or not os.path.exists(landing):
            continue
        with open(landing, encoding="utf-8") as f:
            text = f.read()
        table = render.dashboard_table(t, ents, c)
        text = render.splice_md_region(text, render.DASH_BEGIN, render.DASH_END, table)
        with open(landing, "w", encoding="utf-8") as f:
            f.write(text)
        n_dash += 1

    mc = os.path.join(docs, "mission-control")
    if os.path.isdir(mc):
        with open(os.path.join(mc, "action-center.md"), "w", encoding="utf-8") as f:
            f.write(render.action_center_markdown(c, m))

    with open(os.path.join(project_dir, "corpus.json"), "w", encoding="utf-8") as f:
        json.dump(render.corpus_to_dict(c), f, indent=2)

    print(f"zplus gen-derived: {n_dash} dashboard(s), action center, corpus.json")
    return 0


def main(argv=None):
    return gen_derived(os.getcwd())
```
